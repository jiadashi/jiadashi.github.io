		<script type="text/javascript">
			$(function(){
				$("#btn").click(function(){
					属性
					$("#box>span").attr();        //设置或返回被选元素的属性
					$("#box>span").removeAttr("style");     //从被选元素中移除属性 
					元素，节点
					$("#box>span").html("<span>离元旦还有3天</span>");  //返回或设置被选元素的内容   注释：html和text二个看似相同，却是不同的。html返回的是标签内容，而text返回的是标签及元素
					$("#box>span").text("<span>离元旦还有3天</span>");  //方法设置或返回被选元素的文本内容
					文档处理
					$("#box>span").append("<ul><li>元旦快乐</li></ul>"); //被选元素的结尾（仍然在内部）插入指定内容
					$("<ul><li>元旦快乐</li></ul>").appendTo("#box>span");   //被选元素的结尾（仍然在内部）插入指定内容
					$("#box>span").prepend("<div>警报，警报</div>");      //被选元素的开头（仍位于内部）插入指定内容
					$("<div>警报，警报</div>").prependTo("#box>span");    //被选元素的开头（仍位于内部）插入指定内容
					$("#box>span").after("<p>这是段落标签</p>");         //被选元素后插入指定的内容
					$("<p>这是段落标签</p>").insertAfter("#box>span");     //被选元素之后插入 HTML 标记或已有的元素
					CSS类操作
					$("#box>span").addClass("cen");         //方法向被选元素添加一个或多个类
					$("#box>span").removeClass("cen");      //被选元素移除一个或多个类
				});
			});
		</script>

		<script type="text/javascript">
			$(function(){  
				$("button").click(function(){
					显示隐藏
					$("#box>ul").hide();      ////被选的元素已被显示，则隐藏该元素
					$("#box>ul").show();      ////被选元素已被隐藏，则显示这些元素
					滑动
					$("#box>ul").slideDown(); //通过使用滑动效果，显示隐藏的被选元素
					$("#box>ul").slideToggle();    //通过使用滑动效果（高度变化）来切换元素的可见状态			
					$("#box>ul").slideUp(3000);   //通过使用滑动效果，隐藏被选元素，如果元素已显示出来的话
					淡入淡出
					$("#box>ul").fadeOut();       //使用淡出效果来隐藏被选元素，假如该元素是隐藏的
					$("#box>ul").fadeIn();        //使用淡入效果来显示被选元素，假如该元素是隐藏的
					$("#box>ul").fadeToggle();   //淡入和淡出的切换
					$("#box>ul").fadeTo();        //将被选元素的不透明度逐渐地改变为指定的值
				});
			});
		</script>
