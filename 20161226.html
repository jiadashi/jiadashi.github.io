<!--过滤选择器选择第一个     first-->

		<input type="text"  id="box" value="" />
		<input type="text"  id="btn" value="" />
		<input type="text"  id="bmt" value="" />
		<input type="button" id="an" value="来点我呀！" />
		<script type="text/javascript">
			$(function(){                                         
				$("#an").click(function(){
					$("input:first").css("border","2px solid blue");
				});
			});
		</script>
		:last  过滤并选择出最后一个匹配元素
		:odd  索引为奇数（从0开始计数）的所有元素
		:even  索引为偶数（从0开始计数）的所有元素
		
		
		
<!--过滤选择器选择第几个     eq()-->

		<input type="text"  id="box" value="" />
		<input type="text"  id="btn" value="" />
		<input type="text"  id="bmt" value="" />
		<input type="button" id="an" value="来点我呀！" />
		<script type="text/javascript">
			$(function(){                                         
				$("#an").click(function(){
					$("input:eq(0)").css("border","2px solid blue");
				});      
			});  
		</script>
		:gt()  从匹配的集合中选择索引大于给定值的所有元素
		:lt()  从匹配的集合中选择索引小于给定值的所有元素
		:not   从匹配的集合中去除与给定选择器匹配的元素
