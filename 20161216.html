		    		box.innerHTML = t + "秒后重新获取";
		    		box.disabled = true;                             //当我鼠标点下的时候，按钮就是禁用，不能在次点击
		    		siv1 = window.setInterval(moverr,1000);
		    	}
		    	
		    	function moverr(){
		    		box.innerHTML = t +"秒后重新获得";
		    		if (t<=0) {
		    			if(siv1){
		    				clearInterval(siv1);
		    			}
		    			t = 10;
		    			box.innerHTML = "获取手机验证码";
		    			box.disabled = false;
		    		}
		    		t--;
		    	}
