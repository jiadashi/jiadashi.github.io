jQuery小栗子 动画

联动

			原理：当鼠标移入的时候顶部出现红条，指当前在这个位置，下面的图片跟着切换到移入的框
			$(function(){
         		$(".tab li").mouseenter(function(){     //当鼠标移入的时候
         			$(this).addClass("active").siblings().removeClass("active");   //siblings() 获得匹配集合中每个元素的同胞，通过选择器进行筛选是可选的
         			//removeClass移除，它的样式
         			var idx = $(this).index();  //index 方法返回指定元素相对于其他指定元素的 index 位置.
         			$(".main").eq(idx).addClass("selected").siblings(".main").removeClass("selected");
         			//eq() 选择index()带有指定 的元素
         		})
         	});

图片点击轮播

			原理：当鼠标移入的时候是原来的位置变大，指当前在这个位置，上面的图片跟着切换到数字的位置
			$(function(){
			var a = 1
			$(".wrap >ol >li").mouseenter(function(){
				$(this).addClass("active").siblings().removeClass("active")
				a++;
				var idx = $(this).index()
				$(".wrap >ul >li").eq(idx).css({
					"left":650,
					"z-index":a     //调高层级，移到谁就谁显示
				}).animate({        //animate自定义动画
					"left":0
				},300)
			})
		});

图片滚动

			原理：当鼠标移入的时候，指当前在这个位置，中间的图片跟着切换到指定的图片
			   $(function(){
		            $("#left li").mouseenter(function(){
			        var idx = $(this).index();
			        $("#center li").eq(idx).show().siblings("li").hide();
		});
	});

相对浏览器

			原理：图片定位，当鼠标滚动的时候，图片跟着跑，但是位置不变
			   $(function() {
				$(window).scroll(function() {   //scroll 适用于所有可滚动的元素和 window对象（浏览器窗口）
					var tp = $(document).scrollTop();
					/*console.log(tp);*/
					$("#al img").stop().animate({
						"top": tp
					}, 500)
				})

			});
